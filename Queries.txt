CREATE TABLE IF NOT EXISTS ITEM (ITEM_ID INT AUTO_INCREMENT NOT NULL, NAME VARCHAR(50), MODEL_NUM VARCHAR(25), BATTERIES_REQUIRED BOOLEAN, WEIGHT INT, MANUFACTURING_DATE DATE, PRIMARY KEY (ITEM_ID));

CREATE TABLE IF NOT EXISTS ITEM_REVIEW (REVIEW_ID INT AUTO_INCREMENT NOT NULL, ITEM_ID INT, REVIEW_COMMENT TEXT, REVIEW_DATE DATE, PRIMARY KEY (REVIEW_ID), FOREIGN KEY(ITEM_ID) REFERENCES ITEM (ITEM_ID));

CREATE TABLE IF NOT EXISTS ITEM_REVIEW (REVIEW_ID INT AUTO_INCREMENT NOT NULL, ITEM_ID INT, PRIMARY KEY (REVIEW_ID), FOREIGN KEY(ITEM_ID) REFERENCES ITEM (ITEM_ID));
